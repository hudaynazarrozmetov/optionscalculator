    <script>
        function formatCurrency(input) {
            let value = input.value.replace(/[^0-9.]/g, '');
            if (value) {
                const num = parseFloat(value);
                if (!isNaN(num)) {
                    input.value = '$' + num.toLocaleString('en-US', {minimumFractionDigits: 0, maximumFractionDigits: 2});
                }
            }
        }

        function parseCurrency(value) {
            return parseFloat(value.replace(/[^0-9.]/g, '')) || 0;
        }

        document.getElementById('account').addEventListener('blur', function() { formatCurrency(this); });
        document.getElementById('entryPrice').addEventListener('blur', function() { formatCurrency(this); });

        function calculate() {
            const entryPrice = parseCurrency(document.getElementById('entryPrice').value);
            const contracts = parseFloat(document.getElementById('contracts').value) || 0;
            const stopLossPercent = parseFloat(document.getElementById('stopLossPercent').value) || 0;
            const target1Percent = parseFloat(document.getElementById('target1Percent').value) || 0;
            const target2Percent = parseFloat(document.getElementById('target2Percent').value) || 0;
            const target3Percent = parseFloat(document.getElementById('target3Percent').value) || 0;

            const stopLoss = stopLossPercent > 0 ? entryPrice * (1 - stopLossPercent / 100) : 0;
            const target1 = target1Percent > 0 ? entryPrice * (1 + target1Percent / 100) : 0;
            const target2 = target2Percent > 0 ? entryPrice * (1 + target2Percent / 100) : 0;
            const target3 = target3Percent > 0 ? entryPrice * (1 + target3Percent / 100) : 0;

            const multiplier = 100;
            const totalRisk = contracts * (Math.abs(entryPrice - stopLoss) * multiplier);

            document.getElementById('riskAmount').textContent = '$' + (totalRisk || 0).toFixed(2);
            document.getElementById('stopLossPrice').textContent = '$' + (stopLoss || 0).toFixed(2);
            document.getElementById('target1Price').textContent = '$' + (target1 || 0).toFixed(2);

            const targetsContainer = document.getElementById('targetsContainer');
            targetsContainer.innerHTML = '';

            [
                { price: target1, label: 'Target 1' },
                { price: target2, label: 'Target 2' },
                { price: target3, label: 'Target 3' }
            ].forEach(t => {
                if (t.price > 0) {
                    const profit = contracts * (Math.abs(t.price - entryPrice) * multiplier);
                    const div = document.createElement('div');
                    div.className = 'target-card';
                    div.innerHTML = `<div class="target-title">${t.label}</div><div class="target-profit">$${profit.toFixed(2)}</div>`;
                    targetsContainer.appendChild(div);
                }
            });
        }
    </script>
